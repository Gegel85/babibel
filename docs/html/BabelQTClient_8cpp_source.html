70  std::string password = this->_password.getPlainText();
71  if (username.empty() || password.empty()) {
72  this->_lastError.setText("Last Error: Need valid username and password");
73  return;
74  }
75  if (username.length() > 32 || password.length() > 32) {

77  return;
