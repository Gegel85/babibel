77  void TcpClient::_handleServerPacket(Babel::Network::Protocol::Packet &packet)
78  {
79  this->_lastResponse = std::pair<unsigned char, std::string>{packet.op, packet.data};
80  switch (packet.op) {
81  case Network::Protocol::HELLO:
82  if (packet.data.size() != 2)
83  return this->disconnectFromServer(Network::Protocol::ErrorReason::BAD_PACKET);
84  std::cout << "Server version is ";
85  std::cout << static_cast<unsigned>(static_cast<unsigned char>(packet.data[0] << 8U) + static_cast<unsigned char>(packet.data[1]));
86  std::cout << std::endl;
87  if (packet.data != VERSION_STR)
88  return this->disconnectFromServer(Network::Protocol::ErrorReason::BAD_VERSION);
89  break;
90  case Network::Protocol::BYE:
91  this->disconnectFromServer(Network::Protocol::ErrorReason::NORMAL_CLOSURE);
92  throw Exceptions::DisconnectedException("The server closed the connection: " + Network::Protocol::ErrorReason::errorReasonToString(packet.data));
93  case Network::Protocol::CALL:
94  return this->sendPacketToServer(Network::Protocol::CALL_ACCEPTED, "");
95  case Network::Protocol::CALL_ACCEPTED:
96  if (Network::Protocol::Packet::uint32FromByteString(packet.data))
97  return this->connectToVoice(
98  inet_ntoa({
99  Network::Protocol::Packet::uint32FromByteString(packet.data)
100  }),
101  Network::Protocol::Packet::uint16FromByteString(packet.data.substr(4, 2))
102  );
103  return this->hostVoice(Network::Protocol::Packet::uint16FromByteString(packet.data.substr(4, 2)), 10000);
104  case Network::Protocol::CALL_REFUSED:
105  case Network::Protocol::OK:
106  case Network::Protocol::KO:
107  return;
108  default:
109  return this->disconnectFromServer(Network::Protocol::ErrorReason::BAD_OPCODE);
110  }
